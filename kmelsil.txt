import pandas as pd
import numpy as np
import matplotlib.pyplot as plt
from sklearn.cluster import KMeans
from sklearn.metrics import silhouette_score
from sklearn.preprocessing import StandardScaler

# Assume you already have df and numeric X
df = pd.read_csv("your_dataset.csv")
X = df.select_dtypes(include=[np.number])

# Scale the data (standard practice)
scaler = StandardScaler()
X_scaled = scaler.fit_transform(X)

# 1. Elbow method: try k from 2 to max_k
max_k = 10
k_range = range(2, max_k + 1)
wcss = []                # within‑cluster sum of squares
sil_scores = []          # silhouette scores

for k in k_range:
    kmeans = KMeans(n_clusters=k, init="k-means++", n_init=10, random_state=42)
    labels = kmeans.fit_predict(X_scaled)
    
    wcss.append(kmeans.inertia_)
    sil_scores.append(silhouette_score(X_scaled, labels))

# Plot elbow curve
plt.figure(figsize=(12, 5))

plt.subplot(1, 2, 1)
plt.plot(k_range, wcss, marker="o")
plt.title("Elbow Method")
plt.xlabel("Number of clusters (k)")
plt.ylabel("WCSS")
plt.grid(True)

# Plot silhouette scores
plt.subplot(1, 2, 2)
plt.plot(k_range, sil_scores, marker="o")
plt.title("Silhouette Score vs k")
plt.xlabel("Number of clusters (k)")
plt.ylabel("Silhouette Score")
plt.grid(True)

plt.tight_layout()
plt.show()

# 2. Optionally: pick best k by silhouette
best_k = k_range[np.argmax(sil_scores)]
print("Best k by silhouette score:", best_k)

# 3. Final clustering with that k
final_kmeans = KMeans(n_clusters=best_k, init="k-means++", random_state=42)
final_labels = final_kmeans.fit_predict(X_scaled)

# Add cluster label to df
df["cluster"] = final_labels

print("Final cluster labels (first 10):")
print(df["cluster"].head(10))
