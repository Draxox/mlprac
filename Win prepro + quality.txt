# 1(a) Load data
import pandas as pd

df = pd.read_csv("wine.csv", sep=';')   # or full path
print(df.head())

# 1(b) Check and remove rows with null
print(df.isnull().sum())
df.dropna(inplace=True)  # axis=0 already default

# 1(c) Add new quality category column
def map_quality(x):
    if x <= 4:
        return "Bad"
    elif 5 <= x <= 7:
        return "Average"
    else:
        return "Good"

df["quality_grade"] = df["quality"].apply(map_quality)
